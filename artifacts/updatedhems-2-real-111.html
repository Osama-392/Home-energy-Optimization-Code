class HEMSEnvRealWeather(gym.Env):
    def __init__(self, load_path='House31.csv', solar_path='solar.csv'):
        super(HEMSEnvRealWeather, self).__init__()

        self.load_df = pd.read_csv(load_path, parse_dates=['Date_Time'])
        self.solar_df = pd.read_csv(solar_path, parse_dates=['time'], low_memory=False)
        

        # Remove non-numeric and resample
        self.load_df = self.load_df[['Date_Time', 'Usage_kW']].dropna()
        self.load_df = self.load_df.set_index('Date_Time').resample('10min').mean().reset_index()

        self.solar_df = self.solar_df.set_index('time').resample('10min').mean(numeric_only=True).reset_index()

        # Drop rows with NaN
        self.load_df.dropna(inplace=True)
        self.solar_df.dropna(inplace=True)

        # Sync data lengths
        min_len = min(len(self.load_df), len(self.solar_df))
        self.load_df = self.load_df[:min_len]
        self.solar_df = self.solar_df[:min_len]

        print(f"Final load_df shape: {self.load_df.shape}")
        print(f"Final solar_df shape: {self.solar_df.shape}")

        self.max_battery = 5.0
        self.battery_soc = 2.5
        self.panel_area = 4.0
        self.panel_eff = 0.18
        self.time_step = 0

        self.action_space = spaces.Discrete(4)
        self.observation_space = spaces.Box(low=0, high=1, shape=(7,), dtype=np.float32)

    def _get_obs(self):
        solar_row = self.solar_df.iloc[self.time_step]
        load_row = self.load_df.iloc[self.time_step]
        temp = np.clip((solar_row['air_temperature'] - 20) / 20, 0, 1)
        humidity = np.clip(solar_row['relative_humidity'] / 100, 0, 1)

        return np.array([
            self.time_step / len(self.load_df),
            self.battery_soc / self.max_battery,
            self.solar_output / 5.0,
            self.load_demand / 5.0,
            self.ups_status,
            temp,
            humidity
        ], dtype=np.float32)

    def reset(self, seed=None, options=None):
        super().reset(seed=seed)
        self.time_step = 0
        self.battery_soc = 2.5
        self.ups_status = 1
        self.load_demand = self.load_df.iloc[0]['Usage_kW']
        ghi = self.solar_df.iloc[0]['ghi_pyr']
        cloud_factor = 1 - (self.solar_df.iloc[0].get('cloud_pct', 0) / 100)
        self.solar_output = (ghi * self.panel_area * self.panel_eff * cloud_factor) / 1000
        return self._get_obs(), {}

    def step(self, action):
        if self.time_step >= len(self.load_df):
        # Environment is already done; return dummy values
            dummy_obs = np.zeros(self.observation_space.shape, dtype=np.float32)
            return dummy_obs, 0.0, True, False, {}

        load_row = self.load_df.iloc[self.time_step]
        solar_row = self.solar_df.iloc[self.time_step]

        self.load_demand = load_row['Usage_kW']
        ghi = solar_row['ghi_pyr']
        cloud_factor = 1 - (solar_row.get('cloud_pct', 0) / 100)
        self.solar_output = (ghi * self.panel_area * self.panel_eff * cloud_factor) / 1000

        self.ups_status = 1 if self.battery_soc > 3.0 else 0
        cost = 0

        if action == 0:  # Solar
            if self.solar_output >= self.load_demand:
               excess = self.solar_output - self.load_demand
               chargeable = min(excess, self.max_battery - self.battery_soc)
               self.battery_soc += chargeable
            else:
                 cost = (self.load_demand - self.solar_output) * 0.3

        elif action == 1:  # Battery
            used = min(self.battery_soc, self.load_demand)
            self.battery_soc -= used
            cost = (self.load_demand - used) * 0.3

        elif action == 2:  # UPS
            cost = 0 if self.ups_status else self.load_demand * 0.5

        elif action == 3:  # Grid
            cost = self.load_demand * 0.6

        reward = -cost

        self.time_step += 1
        terminated = self.time_step >= len(self.load_df)
        truncated = False

        if terminated:
            obs = np.zeros(self.observation_space.shape, dtype=np.float32)
        else:
            obs = self._get_obs()

        return obs, reward, terminated, truncated, {}

    def render(self):
        pass

# === Evaluate Trained PPO Agent ===
env = HEMSEnvRealWeather()
obs, _ = env.reset()
done = False

rewards, actions, socs, times = [], [], [], []
costs, action_names = [], []
action_costs = {0: 0, 1: 0, 2: 0, 3: 0}
action_counts = {0: 0, 1: 0, 2: 0, 3: 0}
action_labels = {0: "Solar", 1: "Battery", 2: "UPS", 3: "Grid"}

while not done:
    action = int(model.predict(obs)[0])
    obs, reward, done, _, _ = env.step(action)

    cost = -reward  # because reward = -cost

    rewards.append(reward)
    actions.append(action)
    socs.append(obs[1] * env.max_battery)
    times.append(env.time_step)
    costs.append(cost)

    action_costs[action] += cost
    action_counts[action] += 1
    action_names.append(action_labels[action])

# === Print Cost Summary ===
total_cost = sum(costs)
avg_cost_per_day = total_cost / (len(costs) / (24 * 6))  # 6 steps/hour

print("⚙️  === HEMS Cost Summary ===")
print(f"💰 Total Cost: {total_cost:.2f} PKR")
print(f"📅 Avg Cost per Day: {avg_cost_per_day:.2f} PKR")
print(f"🔢 Total Steps: {len(costs)}")

print("\n🔍 Breakdown by Source:")
for i in range(4):
    label = action_labels[i]
    print(f"• {label}: used {action_counts[i]} times, cost = {action_costs[i]:.2f} PKR")
